#There are also many events that affect Mandate.


#The emperor may pass a series of five Celestial Reforms to better govern China. Passing a reform requires at least 80 Mandate, and lowers Mandate by -70 as well as stability by -1.[8] The reforms must be passed in order, though unlike the Holy Roman Empire's reforms, there is no special reward for completing all of them. If a new nation takes the Mandate, all reforms are lost and Mandate is reset to 60.[9]

chinese_flavor.6 = {
    type = country_event
    hidden = yes

    trigger = {
        trigger_if = {
            limit = { has_global_variable = china_war_count }
            global_var:china_war_count < 4 #max of 3 of these can happen
        }
    }

    immediate = {
        if = {
            limit = {
                trigger_if = {
                    limit = { exists = c:QII }
                    c:QII = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:ZHA }
                    c:ZHA = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:CHU }
                    c:CHU = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:QIN }
                    c:QIN = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:YAN }
                    c:YAN = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:WEI }
                    c:WEI = { war = no }
                }
                trigger_else_if = {
                    limit = { exists = c:HAN }
                    c:HAN = { war = no }
                }
            }
            random_list = {
                1 = {
                    trigger = { exists = c:QII }
                    declare_china_war_effect = { COUNTRY = c:QII }
                }
                1 = {
                    trigger = { exists = c:ZHA }
                    declare_china_war_effect = { COUNTRY = c:ZHA }
                }
                1 = {
                    trigger = { exists = c:CHU }
                    declare_china_war_effect = { COUNTRY = c:CHU }
                }
                1 = {
                    trigger = { exists = c:QIN }
                    declare_china_war_effect = { COUNTRY = c:QIN }
                }
                1 = {
                    trigger = { exists = c:YAN }
                    declare_china_war_effect = { COUNTRY = c:YAN }
                }
                1 = {
                    trigger = { exists = c:WEI }
                    declare_china_war_effect = { COUNTRY = c:WEI }
                }
                1 = {
                    trigger = { exists = c:HAN }
                    declare_china_war_effect = { COUNTRY = c:HAN }
                }
            }
            if = {
                limit = {
                    NOT = { has_global_variable = china_war_count }
                }
                set_global_variable = {
                    name = china_war_count
                    value = 1
                }
            }
            else = {
                change_global_variable = {
                    name = china_war_count
                    add = 1
                }
            }
        }
    }
    after = {
        random_list = {
            1 = {
                trigger_event = {
                    id = chinese_flavor.6
                    days = { 365 700 }
                }
            }
            1 = {
                trigger_event = {
                    id = chinese_flavor.6
                    days = { 700 1200 }
                }
            }
        }
    }
}

chinese_flavor.8 = {
    type = country_event
    hidden = yes

    trigger = {
        trigger_if = {
            limit = { has_global_variable = steppe_war_count }
            global_var:steppe_war_count < 4 #max of 3 of these can happen
        }
    }

    immediate = {
        if = {
            limit = { war = no }
            random_neighbour_country = {
                limit = {
                    is_ai = yes
                    NOT = {
                        alliance_with = root
                    }
                }
                weight = {
                    modifier = {
                        factor = {
                            value = 1
                            add = num_of_cities
                            multiply = 2
                        }
                    }
                }
                prev = {
                    declare_war_with_wargoal = {
                        war_goal = horde_conquest
                        target = prev
                    }
                }
            }
            if = {
                limit = {
                    NOT = {
                        has_global_variable = steppe_war_count
                    }
                }
                add_treasury = 900
                add_manpower = 15
                create_unit = {
                    navy = no
                    location = capital_scope
                    while = {
                        count = 3
                        add_subunit = light_infantry
                    }
                    while = {
                        count = 7
                        add_subunit = light_cavalry
                    }                                          
                    add_food = 100
                    add_morale = 2
                }
                set_global_variable = {
                    name = steppe_war_count
                    value = 1
                }
            }
            else = {
                add_manpower = 15
                add_treasury = 150
                add_war_exhaustion = -10
                create_unit = {
                    navy = no
                    location = capital_scope
                    while = {
                        count = 2
                        add_subunit = light_infantry
                    }
                    while = {
                        count = 4
                        add_subunit = light_cavalry
                    }                                          
                    add_food = 100
                    add_morale = 2
                }
                change_global_variable = {
                    name = steppe_war_count
                    add = 1
                }
            }
        }
    }
    after = {
        trigger_event = {
            id = chinese_flavor.8
            days = { 720 1000 }
        }
    }
}
