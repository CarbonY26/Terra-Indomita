me_germania_scandia_ti.600 = {
    type = country_event

    title = me_germania_scandia_ti_600_title
    desc = me_germania_scandia_ti_600_desc
    picture = great_library 
	left_portrait = root.current_ruler
	right_portrait = scope:scandian_colony_country.current_ruler
	
	immediate = {
		region:caledonia_region = {
			random_region_province = {
				limit = {
					OR = {
						is_in_area = caledonia_meridionalis_area
						is_in_area = caledonia_septentrionalis_area
						is_in_area = caledonia_occidentalis_area
					}
					has_owner = no
					is_coastal = yes
				}
				save_scope_as = scandian_colony
			}
		}
		random_owned_province = {
			limit = {
				any_pops_in_province = {
					pop_culture = root.culture
				}
			}
			save_scope_as = origin_province
		}
		scope:origin_province = {
			random_pops_in_province = {
				limit = {
					pop_culture = root.culture
				}
				move_pop = scope:scandian_colony
			}
		}
		hidden_effect = {
			scope:scandian_colony = {
				create_country = {
					change_government = tribal_chiefdom
					set_primary_culture = root.culture
					set_country_religion = germanic_religion
					save_scope_as = scandian_colony_country
				}
			}
			scope:scandian_colony_country.current_ruler = {
				set_character_religion = germanic_religion
				set_culture = root.culture
			}
		}
	}

    option = {
        name = "me_germania_scandia_ti_600.a"
		add_treasury = -50
		root = {
			make_subject = {
				target = scope:scandian_colony_country
				type = tributary
			}
		}
    }
}
